Zadatak 17, tip: sqrt()

int Korijen(int a)
{
    int i;
    for (int i = 1; i * i <= a; i++)
        if (i * i == a)
            return i;
    return -1;
}


Zadatak 13
int Dec2Bin(short dec), tip: shift operator
{
    int ret, power = 1, i = 0;
    for (i = 1; i <= dec; i *= 2)
    {
        if (dec & i)
            ret += power;
        power *= 10;
    }
    return ret;
}

Zadatak 7
void Sortiraj (double niz[], int n)
{
    for (int i = 0; i < n - 1; i++)
        for (int j = i + 1; i < n; i++)
            if (niz[i] > niz[j])
            {
                temp   = niz[i];
                niz[i] = niz[j];
                niz[j] = temp;
            }
}
